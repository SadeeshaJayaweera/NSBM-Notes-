public class For_While_Loop {
    public static void main(String[] args)
    {
        for(int i=0;i<5;i++)
        {
            System.out.println("Executing Loop:  " +i);
        }

        System.out.println("");

        // Same program in the While Loop
        int i=0;
        while(i<5)
        {
            System.out.println("Executing Loop:  " +i);
            i++;
        }

    }
}
